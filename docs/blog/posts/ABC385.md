        int N, ans = 1;
        cin >> N;
        vi H(N);
        rep(i, N) cin >> H[i];
        for (int d = 1; d < N; d++) {
            for (int s = 0; s < d; s++) {
                int cnt = 0;
                int h = -1;
                for (int j = s; j < N; j += d) {
                    if (h == H[j])
                        cnt++;
                    else
                        cnt = 1;
                    h = H[j];
                    chmax(ans, cnt);
                }
            }
        }
        cout << ans << endl;
    }

    void solve() {
        int N, M, ans = 0;
        ll x, y;
        cin >> N >> M >> x >> y;
        map<ll, set<ll>> xy, yx;
        rep(i, N) {
            ll X, Y;
            cin >> X >> Y;
            xy[X].insert(Y);
            yx[Y].insert(X);
        }

        // (a,b)<->(a,b+d)の道中の家を訪れる
        auto f = [&](map<ll, set<ll>> &ab, map<ll, set<ll>> &ba, ll d, ll a, ll b) -> void {
            ll b2 = b + d;
            if (b > b2)
                swap(b, b2);
            auto it = ab[a].lower_bound(b);
            while (it != ab[a].end() && *it <= b2) {
                ans++;
                ba[*it].erase(a);
                it = ab[a].erase(it);
            }
        };
        char dir;
        ll C;
        rep(i, M) {
            cin >> dir >> C;
            ll nx = x, ny = y;
            if (dir == 'U') {
                ny = y + C;
                f(xy, yx, C, x, y);
            } else if (dir == 'D') {
                ny = y - C;
                f(xy, yx, -C, x, y);
            } else if (dir == 'L') {
                nx = x - C;
                f(yx, xy, -C, y, x);
            } else if (dir == 'R') {
                nx = x + C;
                f(yx, xy, C, y, x);
            }
            x = nx, y = ny;
        }
        cout
            << x << " " << y << " " << ans << endl;
    }
