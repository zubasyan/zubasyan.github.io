    A - Equally

    何かいい方法がないか5秒考えたけど、何も思い浮かばず条件分岐でAC！
    解説を見たら条件分岐でした.
    A=B=C or A=B+C or A+B=C or A+C=B
    しかし！A,B,Cをソートすると A+B==CまたはA=B=Cだけ考えればよくなる！

    void solve() {
    int H, W, x, y, cnt = 0;
    cin >> H >> W >> x >> y;
    x--, y--;
    vector<string> S(H);
    rep(i, H) cin >> S[i];
    string T;
    cin >> T;
    for (char dir : T) {
        rep(d, 4) if (dir == LRUD[d]) {
            int nx = x + di4[d];
            int ny = y + dj4[d];
            if (nx < 0 || nx >= H || ny < 0 || ny >= W)
                continue;
            if (S[nx][ny] == '#')
                continue;
            if (S[nx][ny] == '@')
                cnt++;
            S[nx][ny] = '.';
            x = nx, y = ny;
        }
    }
    x++, y++;
    cout << x << " " << y << " " << cnt << endl;
}

    void solve() {
    int N, ans = 1;
    cin >> N;
    vi H(N);
    rep(i, N) cin >> H[i];
    for (int d = 1; d < N; d++) {
        for (int s = 0; s < d; s++) {
            int cnt = 0;
            int h = -1;
            for (int j = s; j < N; j += d) {
                if (h == H[j])
                    cnt++;
                else
                    cnt = 1;
                h = H[j];
                chmax(ans, cnt);
            }
        }
    }
    cout << ans << endl;
}

    void solve() {
    int N, M, ans = 0;
    ll x, y;
    cin >> N >> M >> x >> y;
    map<ll, set<ll>> xy, yx;
    rep(i, N) {
        ll X, Y;
        cin >> X >> Y;
        xy[X].insert(Y);
        yx[Y].insert(X);
    }

    // (a,b)<->(a,b+d)の道中の家を訪れる
    auto f = [&](map<ll, set<ll>> &ab, map<ll, set<ll>> &ba, ll d, ll a, ll b) -> void {
        ll b2 = b + d;
        if (b > b2)
            swap(b, b2);
        auto it = ab[a].lower_bound(b);
        while (it != ab[a].end() && *it <= b2) {
            ans++;
            ba[*it].erase(a);
            it = ab[a].erase(it);
        }
    };
    char dir;
    ll C;
    rep(i, M) {
        cin >> dir >> C;
        ll nx = x, ny = y;
        if (dir == 'U') {
            ny = y + C;
            f(xy, yx, C, x, y);
        } else if (dir == 'D') {
            ny = y - C;
            f(xy, yx, -C, x, y);
        } else if (dir == 'L') {
            nx = x - C;
            f(yx, xy, -C, y, x);
        } else if (dir == 'R') {
            nx = x + C;
            f(yx, xy, C, y, x);
        }
        x = nx, y = ny;
    }
    cout
        << x << " " << y << " " << ans << endl;
}
